#include <stdio.h>
#include <string.h>

int main() {
    char attendance[10];
    scanf("%s", attendance);
    
    if (strcmp(attendance, "PRESENT") == 0) {
        printf("The student is PRESENT.\n");
    }

    return 0;
}

## Output:

The student is PRESENT.
